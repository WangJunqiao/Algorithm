void dfs(int t){
	if( t==0 ) return;
	
	if( mp.find(t)!=mp.end()){
		if( t==100 || t==1000 ) print("one"); // ****
		print(mp[t]);
	}else{
		if( t>=1000 ){
			dfs(t/1000);
			print("thousand");
			dfs(t%1000);
		}else if( t>=100 ){
			dfs(t/100);
			print("hundred");
			dfs(t%100);
		}else if( t>10 ){
			dfs(t/10*10);
			dfs(t%10);
		}
	}
}
void dfst(int t){
	if( t==0 ) return;
	if( mpt.find(t) !=mpt.end()){
		if( t==100 || t==1000 ) print("one"); // ****
		print(mpt[t]);
	}else{
		if( t>=1000 ){
			dfs(t/1000);
			print("thousand");
			if( t%1000==0 ) { print("th"); return; }
			dfst(t%1000);
		}else if( t>=100 ){
			dfs(t/100);
			print("hundred");
			if( t%100==0 ) { print("th"); return; }
			dfst(t%100);
		}else if( t>10 ){
			dfs(t/10*10);
			dfst(t%10);
		}
	}
}
